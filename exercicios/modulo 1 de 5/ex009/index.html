            using System;
            using System.Threading;
            namespace Teste
            {
                public static class TesteCode
                {
                    public static void Main(String[] args)
                    {
                        String Answer;
                        Console.WriteLine("Olá, Mundo. Esse é um Teste! Deseja TESTAR? S/N ");
                        Answer = Console.ReadLine();
                        if (Answer == "S")
                        {
        
                            Console.WriteLine("Testando...");
                            System.Threading.Thread.Sleep(2000);
                            Console.WriteLine("Testando..");
                            System.Threading.Thread.Sleep(2000);
                            Console.WriteLine("Testando.");
                            System.Threading.Thread.Sleep(2000);
                            Console.Clear();
                            Console.WriteLine("Teste CONCLUIDO com SUCESSO!!");
                            System.Threading.Thread.Sleep(2000);
                            Console.Clear();
                            Console.WriteLine("Todos os sistemas OPERANTES!!");
        
                        }
                        else
                        {
                            Console.Clear();
                            Console.WriteLine("Okay, Sistema não será testado!!"")
                        }
                         Console.WriteLine("\nPressione qualquer tecla para sair...");
                         Console.ReadKey();
                    }
                }
            }
